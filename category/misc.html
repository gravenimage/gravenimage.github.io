Tue 25 August 2015

Gearing up to a pub quiz tomorrow night, my friends and I started to have a facebook conversation only using words that could be made from the symbols of the periodic table. (Please don't comment) I spied a chance to ignore my wife for half-an-hour and impress her with my chemical wit afterwards - surely it can't be difficult to create a quick script to generate all the combinations? Spoiler - no, it's easy!

First challenge - getting a list of elements in a convenient format.

http://pastebin.com/raw.php?i=CKwm136x - a quick google and done!

Then, the script. I've cheated here and replaced the csv-parsing with a set constructor - but I did learn about the next method to advance an iteratable (to skip the header on the CSV file). I guess that without more work than I can squeeze in (i.e. not capable of), I'm not going to do better than Python's set union and intersection methods. Ditto for generating the permutations.

The set of possibles expands very rapidly, and kills my little Chromebook with n >= 5 with out-of-memory. So, with quite a lot of uglification we get a version which doesn't realize any results until as late as possible. It also does a pre-scan of the elements and combinations of two elements and removes them from consideration if they never appear.

import itertools
import unittest


with open('/usr/share/dict/words') as f:
    words = set([s.strip().replace("'", '').lower() for s in f.readlines() if not s.strip().endswith("'s")])
joined_words = ' '.join(words)

elements = set(['uuq', 'ge', 'gd', 'ga', 'la', 'li', 'tl', 'tm', 'lr', 'th', 'ti', 'te', 'tb', 'tc', 'ta', 'yb', 'db', 'dy', 'xe', 'h', 'p', 'zn', 'eu', 'zr', 'er', 'ru', 're', 'rf', 'ra', 'rb', 'rn', 'rh', 'be', 'ba', 'uuu', 'uut', 'bh', 'bi', 'uus', 'bk', 'uuo', 'uun', 'uuh', 'br', 'uub', 'c', 'k', 'o', 's', 'w', 'os', 'co', 'cm', 'cl', 'ca', 'pa', 'cf', 'ce', 'cd', 'cs', 'cr', 'cu', 'pr', 'pt', 'pu', 'pb', 'lu', 'pd', 'po', 'pm', 'hs', 'ho', 'hf', 'hg', 'he', 'md', 'mg', 'b', 'f', 'mo', 'mn', 'n', 'mt', 'v', 'ac', 'ag', 'ir', 'am', 'al', 'as', 'ar', 'au', 'at', 'in', 'ni', 'no', 'na', 'nb', 'nd', 'ne', 'uup', 'es', 'np', 'fr', 'sc', 'fe', 'fm', 'i', 'sr', 'kr', 'si', 'u', 'sn', 'sm', 'y', 'sb', 'sg', 'se'])

# remmove totally impossible elements
elements = [el for el in elements if el in joined_words]

# remove element pairs which are impossible
twoers = set(itertools.product(elements, repeat=2))

imps = None
def impossibles():
    global imps
    if imps is None:
        imps = set([tup for tup in twoers if ''.join(tup) not in joined_words])
    return imps

gts = None
def good_twoers():
    global gts
    if gts is None:
        gts = twoers - impossibles()
    return gts

def is_impossible(tup):
    # short circuit recursion for speed
    if len(tup) < 2: return False
    if len(tup) == 2: return tup in impossibles()
    if len(tup) == 3: return tup[0:2] in impossibles() or tup[1:3] in impossibles()
    if len(tup) == 4: return tup[0:2] in impossibles() or tup[1:3] in impossibles() or tup[2:4] in impossibles()
    if len(tup) == 5: return tup[0:2] in impossibles() or tup[1:3] in impossibles() or tup[2:4] in impossibles() or tup[3:5] in impossibles()
    if len(tup) > 5: return tup[0:2] in impossibles() or is_impossible(tup[1:])

def flatten_tup(tup):
    lst = []
    for x in tup:
        if isinstance(x, tuple):
            for y in x:
                lst.append(y)
        else:
            lst.append(x)
    return tuple(lst)

def flatten(itr):
    for x in itr:
        yield flatten_tup(x)

def words_of(num_els):
    args = [good_twoers()] * (num_els / 2)
    if num_els % 2 == 1:
        args.append(elements)

    combinations = itertools.product(*args)
    return set((word for word in (''.join(c) for c in flatten(combinations) if not is_impossible(c)) if word in words))

def elemental_words(count):
    all_words = set()
    for i in range(2, count + 1):
        all_words = all_words.union(words_of(i))
    return all_words

class SimpleTest(unittest.TestCase):

    def test_flattened(self):
        self.assertEqual((1,2,3,4), flatten_tup(((1,2),3,4)))

    def test_correct_2(self):
        self.assertEqual(532, len(elemental_words(2)))

    def test_correct_3(self):
        self.assertEqual(2269, len(elemental_words(3)))

    def test_correct_4(self):
        self.assertEqual(5040, len(elemental_words(4)))

if __name__ == "__main__":
    import sys
    words = elemental_words(int(sys.argv[1]))
    print words, len(words)
